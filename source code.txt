#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Node structure
typedef struct Book {
    int studentID;
    int bookID;
    char title[50];
    char date[15];
    struct Book* next;
} Book;

Book* head = NULL;

// Function to issue a new book
void issueBook(int studentID, int bookID, char title[], char date[]) {
    Book* newBook = (Book*)malloc(sizeof(Book));
    newBook->studentID = studentID;
    newBook->bookID = bookID;
    strcpy(newBook->title, title);
    strcpy(newBook->date, date);
    newBook->next = head;
    head = newBook;
    printf("Book issued successfully!\n");
}

// Function to return (delete) a book
void returnBook(int bookID) {
    Book *temp = head, *prev = NULL;
    while (temp != NULL && temp->bookID != bookID) {
        prev = temp;
        temp = temp->next;
    }
    if (temp == NULL) {
        printf("Book not found!\n");
        return;
    }
    if (prev == NULL) head = temp->next;
    else prev->next = temp->next;
    free(temp);
    printf("Book returned successfully!\n");
}

// Search by student ID
void searchByStudent(int studentID) {
    Book* temp = head;
    int found = 0;
    while (temp != NULL) {
        if (temp->studentID == studentID) {
            printf("BookID: %d | Title: %s | Date: %s\n", temp->bookID, temp->title, temp->date);
            found = 1;
        }
        temp = temp->next;
    }
    if (!found) printf("No books found for Student ID %d\n", studentID);
}

// Display all borrowed books
void displayBooks() {
    Book* temp = head;
    if (temp == NULL) {
        printf("No borrowed books.\n");
        return;
    }
    printf("\nBorrowed Books:\n");
    while (temp != NULL) {
        printf("StudentID: %d | BookID: %d | Title: %s | Date: %s\n",
               temp->studentID, temp->bookID, temp->title, temp->date);
        temp = temp->next;
    }
}

// Count total books
void countBooks() {
    Book* temp = head;
    int count = 0;
    while (temp != NULL) {
        count++;
        temp = temp->next;
    }
    printf("Total borrowed books: %d\n", count);
}

int main() {
    int choice, studentID, bookID;
    char title[50], date[15];

    while (1) {
        printf("\n--- Library Book Borrowing System ---\n");
        printf("1. Issue Book\n2. Return Book\n3. Search by Student ID\n4. Display All Borrowed Books\n5. Count Borrowed Books\n6. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter Student ID, Book ID, Title, Date: ");
                scanf("%d %d %s %s", &studentID, &bookID, title, date);
                issueBook(studentID, bookID, title, date);
                break;
            case 2:
                printf("Enter Book ID to return: ");
                scanf("%d", &bookID);
                returnBook(bookID);
                break;
            case 3:
                printf("Enter Student ID: ");
                scanf("%d", &studentID);
                searchByStudent(studentID);
                break;
            case 4:
                displayBooks();
                break;
            case 5:
                countBooks();
                break;
            case 6:
                exit(0);
            default:
                printf("Invalid choice!\n");
        }
    }
}
